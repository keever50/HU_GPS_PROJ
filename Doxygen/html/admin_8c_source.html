262void CreateTasks(void)
263{
264 PTASKDATA ptd = tasks; // tasks == &tasks[0]: both are addresses of first struct
265 vTaskSuspendAll(); // zet de scheduler stil en dwing FreeRTOS om eerst alle taken te starten
266
267 for (; ptd->func != NULL; ptd++) // loopje zonder teller; gewoon de pointer steeds ophogen...
268 {
269 UART_puts("\r\nCreated: ");
270 UART_puts((char *)ptd->attr.name);
271
272 // nb. kijk naar de haakjes van argument 3: het gaat om het adres van (structure attr).
273 ptd->hTask = osThreadNew(ptd->func, ptd->argument, &(ptd->attr));
274 }
275
276 xTaskResumeAll(); // start nu de scheduler: play ball
277 DisplayTaskData(); // display alle taskdata op UART
278}

337void StartStopTask(int task_id)
338{
339 PTASKDATA ptd = tasks;
340 unsigned int task_nr;
341 TaskStatus_t xTaskDetails;
342
343 for (task_nr = 1; ptd->func != NULL; ptd++, task_nr++)
344 if (task_nr == task_id) // hebbes
345 {
346 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid);
347 UART_puts("\r\n");
348 UART_puts(ptd->attr.name);
349
350 // een task kan meerdere statussen hebben; het wordt simpeler
351 // als je alleen de eSuspended checkt.
352 if (xTaskDetails.eCurrentState == eSuspended)
353 {
354 vTaskResume(ptd->hTask);
355 UART_puts(" resumed.");
356 }
357 else // bv. eBlocked, eRunning, eInvalid etc.
358 {
359 vTaskSuspend(ptd->hTask);
360 UART_puts(" suspended.");
361 }
362 return;
363 }
364}

371void DisplayTaskData(void)
372{
373 PTASKDATA ptd = tasks;
374 UBaseType_t freespace;
375 TaskStatus_t xTaskDetails;
376 unsigned int used;
377 unsigned int task_nr; // tasknr for changing priority
378
379 for (task_nr=1; ptd->func != NULL; ptd++, task_nr++)
380 {
381 freespace = uxTaskGetStackHighWaterMark(ptd->hTask) * 4; // amount of free bytes
382 used = 100 - (freespace * 100) / ptd->attr.stack_size; // in percentage
383
384 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid); // get task status
385
386 UART_puts("\r\n\t["); UART_putint(task_nr); UART_puts("] ");
387 UART_puts(ptd->attr.name);
388 UART_puts("\t priority: "); UART_putint(ptd->attr.priority);
389 UART_puts("\t stacksize: "); UART_putint(ptd->attr.stack_size);
390 UART_puts("\t used: "); UART_putint(used); UART_puts("%");
391 UART_puts("\t status: "); UART_puts(xTaskDetails.eCurrentState == eSuspended ? "suspended": "running");
392 }
393 UART_puts("\r\n");
394}
